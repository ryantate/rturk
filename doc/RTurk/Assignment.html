def initialize(xml_object)
  map_content(xml_object,
    :id => 'AssignmentId',
    :worker_id => 'WorkerId',
    :status => 'AssignmentStatus',
    :accepted_at => 'AcceptTime',
    :approved_at => "ApprovalTime",
    :submitted_at => 'SubmitTime')
    self.answers = xml_object.xpath('Answer/*').to_s
end
